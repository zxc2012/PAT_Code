void dfs(vector<int> adj[], vector<int> dp[], int src){
    for (auto child : adj[src]) {
        // not including source in the vertex cover
        dp[src][0] += dp[child][1];
        // including source in the vertex cover
        dp[src][1] += min(dp[child][1], dp[child][0]);
    }
    dp[src][1] += 1;
}
