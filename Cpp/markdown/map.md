int* last = NULL;//flag
bool isValidBST(TreeNode* root) {
   if (root){
         if(!isValidBST(root->left)) return false;
         if (last && *last>=root->val) return false;//第一次智慧执行下面
         last = &root->val;
         if(!isValidBST(root->right)) return false;
         return true;
   }else return true;
};
